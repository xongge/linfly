local function deobfuscate(s)
    local valid_chars = {}  -- 这里的{}是转义后的写法
    for c in s:gmatch('.') do
        if c:match('[a-zA-Z0-9+/=]') then
            table.insert(valid_chars, c)
        end
    end
    return table.concat(valid_chars)
end

local function b64decode(s)
    local result = ""
    local a, b, c, d
    for i = 1, #s, 4 do
        a = b64_map[s:sub(i, i)] or 0
        b = b64_map[s:sub(i+1, i+1)] or 0
        c = b64_map[s:sub(i+2, i+2)] or 0
        d = b64_map[s:sub(i+3, i+3)] or 0
        result = result .. string.char(bit32.band(bit32.rshift(a*64 + b, 2), 0xFF))
        if i+2 <= #s and c ~= 0 then
            result = result .. string.char(bit32.band(bit32.lshift(b, 4) + bit32.rshift(c, 2), 0xFF))
        end
        if i+3 <= #s and d ~= 0 then
            result = result .. string.char(bit32.band(bit32.lshift(c, 6) + d, 0xFF))
        end
    end
    return result
end

        end
        local function rot_word(word)
            return (word << 8) | (word >> 24)
        end
        local w = {}
        for i = 1, 4 do
            w[i] = (key[(i-1)*4 + 1] << 24) | (key[(i-1)*4 + 2] << 16) | (key[(i-1)*4 + 3] << 8) | key[(i-1)*4 + 4]
        end
        for i = 5, 4*(1+10) do
            local temp = w[i-1]
            if i % 4 == 1 then
                temp = sub_word(rot_word(temp)) ~ 0x01000000
            end
            w[i] = w[i-4] ~ temp
        end
        local state = {
            (block[1] << 24) | (block[2] << 16) | (block[3] << 8) | block[4],
            (block[5] << 24) | (block[6] << 16) | (block[7] << 8) | block[8],
            (block[9] << 24) | (block[10] << 16) | (block[11] << 8) | block[12],
            (block[13] << 24) | (block[14] << 16) | (block[15] << 8) | block[16]
        }
        for i = 1, 4 do
            state[i] = state[i] ~ w[i]
        end
        for round = 1, 9 do
            for i = 1, 4 do
                local a = state[i]
                local b = state[i%4 + 1]
                local c = state[(i+1)%4 + 1]
                local d = state[(i+2)%4 + 1]
                state[i] = a ~ sub_word(b) ~ sub_word(c) ~ sub_word(d) ~ w[round*4 + i]
            end
        end
        local res = {}
        for i = 1, 4 do
            local val = state[i] ~ w[11*4 + i]
            table.insert(res, (val >> 24) & 0xFF)
            table.insert(res, (val >> 16) & 0xFF)
            table.insert(res, (val >> 8) & 0xFF)
            table.insert(res, val & 0xFF)
        end
        return res
    end

    local plaintext = ""
    for i = 1, #ciphertext, 16 do
        local block = {}  -- 转义
        for j = 1, 16 do
            local b = ciphertext:byte(i + j - 1) or 0
            table.insert(block, b)
        end
        local decrypted = aes_decrypt_block(key_bytes, block)
        for _, b in ipairs(decrypted) do
            plaintext = plaintext .. string.char(b)
        end
    end
    local pad_len = plaintext:byte(#plaintext)
    return plaintext:sub(1, #plaintext - pad_len)
end
